   108	                        task: str = "convert") -> LLMProfile:
   109	        """Resolve the active LLM profile.
   110	
   111	        Resolution order (spec §3.5, §6.1):
   112	        1. --llm-profile CLI override (bypasses routing, disables fallback)
   113	        2. routing.<task>.primary
   114	        3. routing.default.primary
   115	        4. first available profile
   116	
   117	        Args:
   118	            override: CLI --llm-profile override (takes precedence).
   119	            task: Task name for route lookup (default: "convert").
   120	
   121	        Returns:
   122	            The resolved LLMProfile.
   123	
   124	        Raises:
   125	            ValueError: if the profile name is not found.
   126	        """
   127	        if override:
   128	            # CLI override: bypass routing entirely
   129	            profile = self.profiles.get(override)
   130	            if profile is None:
   131	                available = ", ".join(sorted(self.profiles.keys()))
   132	                raise ValueError(
   133	                    f"Unknown LLM profile '{override}'. Available: {available}"
   134	                )
   135	            return profile
   136	
   137	        # Route resolution: task route → default route
   138	        route = self.routing.get(task) or self.routing.get("default")
   139	        if route and route.primary:
   140	            profile = self.profiles.get(route.primary)
   141	            if profile:
   142	                return profile
   143	            raise ValueError(
   144	                f"Route '{task}' references missing profile '{route.primary}'. "
   145	                f"Available: {', '.join(sorted(self.profiles.keys()))}"
   146	            )
   147	
   148	        # Fallback: default route
   149	        if task != "default":
   150	            default_route = self.routing.get("default")
   151	            if default_route and default_route.primary:
   152	                profile = self.profiles.get(default_route.primary)
   153	                if profile:
   154	                    return profile

   157	                    f"Available: {', '.join(sorted(self.profiles.keys()))}"
   158	                )

   160	        raise ValueError(

    35	class ScopeOrCommandGroup(click.Group):
    36	    """Click group that accepts either a subcommand or a bare scope argument.
    37	
    38	    v1.2 extension (folio-enrich-diagnose CB-2 closure): handles BOTH
    39	    bare-positional-first (`folio enrich ClientA`) AND option-before-scope
    40	    (`folio enrich --dry-run ClientA`) forms by walking args, recognizing
    41	    Click option-with-value pairs, and rewriting the FIRST non-option,
    42	    non-subcommand token as `--scope X` regardless of position.
    43	
    44	    v1.3 extension (folio-enrich-diagnose CB-2 final closure): once a
    45	    registered subcommand token is encountered, the parser passes the
    46	    subcommand token AND all remaining tokens verbatim — no further
    47	    `--scope` rewrites. Prevents subcommand positional args from being
    48	    rewritten to group-level `--scope`.
    49	    """
    50	
    51	    def parse_args(self, ctx, args):
    52	        # Determine which group-level options take a value (so their value
    53	        # token is not mistaken for a positional scope).
    54	        value_taking_short = set()
    55	        value_taking_long = set()
    56	        for param in self.params:
    57	            if isinstance(param, click.Option) and not param.is_flag and not param.count:
    58	                for opt in param.opts:
    59	                    if opt.startswith("--"):
    60	                        value_taking_long.add(opt)
    61	                    elif opt.startswith("-"):
    62	                        value_taking_short.add(opt)
    63	
    64	        new_args: list[str] = []
    65	        rewrote = False
    66	        skip_next = False
    67	        saw_subcommand = False
    68	        i = 0
    69	        while i < len(args):
    70	            arg = args[i]
    71	            if saw_subcommand:
    72	                # v1.3 closure: pass subcommand args verbatim (do NOT rewrite
    73	                # the subcommand's positional scope to group-level --scope).
    74	                new_args.extend(args[i:])
    75	                break
    76	            if skip_next:
    77	                new_args.append(arg)
    78	                skip_next = False
    79	                i += 1
    80	                continue
    81	            if arg.startswith("-"):
    82	                new_args.append(arg)
    83	                opt_name = arg.split("=", 1)[0]
    84	                if "=" not in arg and (
    85	                    opt_name in value_taking_long or opt_name in value_taking_short
    86	                ):
    87	                    skip_next = True
    88	                i += 1
    89	                continue
    90	            # Non-option token
    91	            if rewrote:
    92	                new_args.append(arg)
    93	                i += 1
    94	                continue
    95	            if self.get_command(ctx, arg) is not None:
    96	                # Registered subcommand; pass through and stop rewriting.
    97	                new_args.append(arg)
    98	                saw_subcommand = True
    99	                i += 1
   100	            else:
   101	                # Treat as scope. Rewrite to --scope X.
   102	                new_args.extend(["--scope", arg])
   103	                rewrote = True
   104	                i += 1
   105	        return super().parse_args(ctx, new_args)

   283	def _classify_outcome(exc: Exception) -> str:
   284	    """Classify an exception into an allowed outcome bucket.
   285	
   286	    Stage 1 additions: recognizes oversized image, model API, PDF
   287	    corruption, and rate-limiting failures for better batch reports.
   288	    """
   289	    msg = str(exc).lower()
   290	    exc_type = type(exc).__name__
   291	
   292	    # Timeout
   293	    if "timed out" in msg or "timeout" in msg:
   294	        return "timeout"
   295	
   296	    # AppleScript error codes (e.g., "error number -9074")
   297	    match = _APPLESCRIPT_ERROR_RE.search(str(exc))
   298	    if match:
   299	        return f"applescript_{match.group(1)}"
   300	
   301	    # Stage 1: Oversized image (Pillow MAX_IMAGE_PIXELS or DPI backoff failure)
   302	    if isinstance(exc, ImageExtractionError) and (
   303	        "too large" in msg or "exceeds limit" in msg
   304	    ):
   305	        return "oversized_image"
   306	    if "decompression bomb" in msg or "max_image_pixels" in msg:
   307	        return "oversized_image"
   308	
   309	    # Stage 1: Model / LLM API errors
   310	    if any(kw in exc_type.lower() for kw in ("apierror", "apiconnection", "apistatus")):
   311	        return "model_error"
   312	    if any(kw in msg for kw in ("api error", "model not found", "invalid api key")):
   313	        return "model_error"
   314	
   315	    # Stage 1: Rate limiting
   316	    if "rate limit" in msg or "429" in msg or "too many requests" in msg:
   317	        return "rate_limited"
   318	
   319	    # Stage 1: PDF corruption / rendering failure
   320	    if isinstance(exc, ImageExtractionError):
   321	        # S-2 fix: distinguish dependency/environment errors
   322	        if "not found" in msg or "install" in msg:
   323	            return "missing_dependency"
   324	        return "pdf_render_error"
   325	    if any(kw in msg for kw in ("corrupt", "invalid pdf", "unable to open")):
   326	        return "pdf_corrupt"
   327	
   328	    # R4-#4: Provider endpoint blocked / misconfigured
   329	    if isinstance(exc, EndpointNotAllowedError):
   330	        return "endpoint_blocked"
   331	
   332	    return "unknown"
