   286	def _parse_slide_boundaries(raw_text: str) -> dict[int, str]:
   287	    """Parse MarkItDown output into per-slide text blocks.
   288	
   289	    MarkItDown typically marks slide boundaries with HTML comments
   290	    like <!-- Slide X --> or with markdown headers like ## Slide X.
   291	    This parser handles both patterns.
   292	    """
   293	    slides = {}
   294	
   295	    # Pattern 1: HTML comments (<!-- Slide number: N --> or <!-- Slide N -->)
   296	    comment_pattern = re.compile(
   297	        r"<!--\s*Slide\s*(?:number:\s*)?(\d+)\s*-->", re.IGNORECASE
   298	    )
   299	    # Pattern 2: Markdown headers (## Slide N or # Slide N)
   300	    header_pattern = re.compile(r"^#{1,3}\s+Slide\s+(\d+)", re.IGNORECASE | re.MULTILINE)
   301	
   302	    # Try comment-based splitting first
   303	    comment_matches = list(comment_pattern.finditer(raw_text))
   304	    if comment_matches:
   305	        return _split_by_matches(raw_text, comment_matches)
   306	
   307	    # Try header-based splitting
   308	    header_matches = list(header_pattern.finditer(raw_text))
   309	    if header_matches:
   310	        return _split_by_matches(raw_text, header_matches)
   311	
   312	    # Pattern 2.5: Horizontal rules (--- or ===) as slide separators
   313	    # Some MarkItDown versions use these between slides
   314	    # Strip leading YAML frontmatter block before HR splitting
   315	    text_for_hr = raw_text
   316	    if text_for_hr.startswith("---"):
   317	        frontmatter_end = text_for_hr.find("---", 3)
   318	        if frontmatter_end != -1:
   319	            candidate = text_for_hr[3:frontmatter_end].strip()
   320	            try:
   321	                import yaml
   322	                parsed = yaml.safe_load(candidate)
   323	                if isinstance(parsed, dict):
   324	                    # Looks like real YAML frontmatter — strip it
   325	                    text_for_hr = text_for_hr[frontmatter_end + 3:]
   326	            except Exception:
   327	                pass  # Not valid YAML — keep the text intact
   328	    hr_pattern = re.compile(r"^-{3,}$|^={3,}$", re.MULTILINE)
   329	    hr_matches = list(hr_pattern.finditer(text_for_hr))
   330	    if len(hr_matches) >= 2:
   331	        # Guard: require minimum content between HRs to avoid false positives
   332	        has_content_between = False
   333	        for j in range(len(hr_matches) - 1):
   334	            between = text_for_hr[hr_matches[j].end():hr_matches[j + 1].start()].strip()
   335	            if len(between) > 10:
   336	                has_content_between = True
   337	                break
   338	        if has_content_between:
   339	            return _split_by_hr(text_for_hr, hr_matches)
   340	
   341	    # Pattern 3: Numbered sections starting from 1 with sequential numbering.
   342	    # Requires minimum content (>20 chars) between numbered markers to avoid
   343	    # false-positives on numbered lists within slide body text.
   344	    section_pattern = re.compile(r"^(\d+)\s*$", re.MULTILINE)
   345	    section_matches = list(section_pattern.finditer(raw_text))
   346	    if len(section_matches) >= 2:
   347	        # Validate sequential numbering from 1
   348	        numbers = [int(m.group(1)) for m in section_matches]
   349	        if numbers[0] == 1 and numbers == list(range(1, len(numbers) + 1)):
   350	            # Guard: require substantial content between numbered markers
   351	            has_substantial_content = False
   352	            for j in range(len(section_matches) - 1):
   353	                between = raw_text[section_matches[j].end():section_matches[j + 1].start()].strip()
   354	                if len(between) > 20:
   355	                    has_substantial_content = True
   356	                    break
   357	            if has_substantial_content:
   358	                return _split_by_matches(raw_text, section_matches)
   359	
   360	    return {}
   361

   363	def _split_by_hr(raw_text: str, hr_matches: list) -> dict[int, str]:
   364	    """Split text into slides using horizontal rules as boundaries."""
   365	    slides = {}
   366	    slide_num = 1
   367	
   368	    # Text before first HR
   369	    first_text = raw_text[:hr_matches[0].start()].strip()
   370	    if first_text:
   371	        slides[slide_num] = first_text
   372	        slide_num += 1
   373	
   374	    # Text between HRs
   375	    for i, match in enumerate(hr_matches):
   376	        start = match.end()
   377	        end = hr_matches[i + 1].start() if i + 1 < len(hr_matches) else len(raw_text)
   378	        text = raw_text[start:end].strip()
   379	        text = re.sub(r"<!--.*?-->", "", text, flags=re.DOTALL).strip()
   380	        if text:
   381	            slides[slide_num] = text
   382	            slide_num += 1
   383	
   384	    return slides

   387	def _split_by_matches(raw_text: str, matches: list) -> dict[int, str]:
   388	    """Split text into slides based on regex matches."""
   389	    slides = {}
   390	    for i, match in enumerate(matches):
   391	        slide_num = int(match.group(1))
   392	        start = match.end()
   393	        end = matches[i + 1].start() if i + 1 < len(matches) else len(raw_text)
   394	        text = raw_text[start:end].strip()
   395	        # Clean up any remaining HTML comments
   396	        text = re.sub(r"<!--.*?-->", "", text, flags=re.DOTALL).strip()
   397	        if text:
   398	            slides[slide_num] = text
   399	    return slides

   402	def reconcile_slide_count(
   403	    slide_texts: dict[int, SlideText],
   404	    image_count: int,
   405	) -> ReconciliationResult:
   406	    """Pad or truncate slide text dict to match authoritative image count.
   407	
   408	    Image count (from PDF page count) is authoritative. Text extraction
   409	    may produce fewer slides (missed boundaries) or more (false boundaries).
   410	
   411	    Keys are NEVER remapped — they preserve page/slide positional identity.
   412	    Gaps within the existing key range are filled with empty SlideText entries.
   413	
   414	    Returns:
   415	        ReconciliationResult with reconciled slide_texts and metadata.
   416	        Consumers should check alignment_confidence to assess trust.
   417	    """
   418	    # B2 fix: copy input to avoid mutating caller's dict
   419	    slide_texts = dict(slide_texts)
   420	
   421	    if not slide_texts:
   422	        # No text at all — return empty placeholders
   423	        return ReconciliationResult(
   424	            slide_texts={
   425	                i: SlideText(slide_num=i, full_text="", is_empty=True)
   426	                for i in range(1, image_count + 1)
   427	            },
   428	            was_reconciled=True,
   429	            action="padded",
   430	            gaps_filled=0,
   431	            original_text_count=0,
   432	            image_count=image_count,
   433	            alignment_confidence=0.0,
   434	        )
   435	
   436	    original_text_count = len(slide_texts)
   437	
   438	    # Step 1: Fill gaps within existing key range with empties
   439	    max_existing_key = max(slide_texts.keys())
   440	    gaps_filled = 0
   441	    for i in range(1, max_existing_key + 1):
   442	        if i not in slide_texts:
   443	            slide_texts[i] = SlideText(slide_num=i, full_text="", is_empty=True)
   444	            gaps_filled += 1
   445	
   446	    if gaps_filled:
   447	        logger.info(
   448	            "Filled %d gaps in text keys (range 1..%d)",
   449	            gaps_filled, max_existing_key,
   450	        )
   451	
   452	    # Step 2: Pad or truncate to match image count
   453	    text_count = len(slide_texts)
   454	    confidence = (
   455	        min(original_text_count, image_count) / max(original_text_count, image_count)
   456	        if image_count > 0 else 0.0
   457	    )
   458	
   459	    if text_count == image_count:
   460	        return ReconciliationResult(
   461	            slide_texts=dict(slide_texts),
   462	            was_reconciled=gaps_filled > 0,
   463	            action="gap_filled" if gaps_filled > 0 else "none",
   464	            gaps_filled=gaps_filled,
   465	            original_text_count=original_text_count,
   466	            image_count=image_count,
   467	            alignment_confidence=confidence,
   468	        )
   469	
   470	    if text_count > image_count:
   471	        logger.warning(
   472	            "Text extraction found %d slides but only %d images. "
   473	            "Truncating text to match image count.",
   474	            text_count, image_count,
   475	        )
   476	        return ReconciliationResult(
   477	            slide_texts={
   478	                k: v for k, v in slide_texts.items() if k <= image_count
   479	            },
   480	            was_reconciled=True,
   481	            action="truncated",
   482	            gaps_filled=gaps_filled,
   483	            original_text_count=original_text_count,
   484	            image_count=image_count,
   485	            alignment_confidence=confidence,
   486	        )
   487	
   488	    # text_count < image_count — pad missing slides
   489	    logger.warning(
   490	        "Text extraction found %d slides but %d images. "
   491	        "Padding %d missing slides with empty text.",
   492	        text_count, image_count, image_count - text_count,
   493	    )
   494	    result = dict(slide_texts)
   495	    for i in range(1, image_count + 1):
   496	        if i not in result:
   497	            result[i] = SlideText(slide_num=i, full_text="", is_empty=True)
   498	    return ReconciliationResult(
   499	        slide_texts=result,
   500	        was_reconciled=True,
   501	        action="padded",
   502	        gaps_filled=gaps_filled,
   503	        original_text_count=original_text_count,
   504	        image_count=image_count,
   505	        alignment_confidence=confidence,
   506	    )

   448	            diagram_extract_slides = slides_to_coerce

   450	            if diagram_extract_slides:
   451	                from .pipeline import diagram_extraction as diag_ext

   499	                        existing_analysis.review_questions = frozen_a.review_questions

   503	            # Stage 4b: Optional depth pass

   753	    def _convert_document(
   754	        self,
   755	        *,
   756	        source_path: Path,
   757	        note: Optional[str],
   758	        client: Optional[str],
   759	        engagement: Optional[str],
   760	        target: Optional[Path],
   761	        subtype: str,
   762	        industry: Optional[list[str]],
   763	        extra_tags: Optional[list[str]],
   764	        llm_profile: Optional[str],
   765	        preserved_enrich_fields: Optional[dict],
   766	    ) -> ConversionResult:
   767	        """Convert a DOCX as a document-oriented single evidence note."""
   768	
   769	        deck_name = _sanitize_name(source_path.stem)
   770	        logger.info("Converting document: %s", source_path.name)

   237	def reconcile_from_frontmatter(library_root: Path, data: dict) -> dict:
   238	    """Reconcile registry entries against their actual markdown frontmatter.
   239	
   240	    Updates registry fields that are frontmatter-authoritative
   241	    (title, type, client, engagement, authority, curation_level,
   242	    review_status, review_flags) so the registry stays consistent
   243	    after manual edits.
   244	    """
   245	    library_root = Path(library_root).resolve()
   246	    changed = 0
   247	    for deck_id, entry_data in list(data.get("decks", {}).items()):
   248	        md_rel = entry_data.get("markdown_path", "")
   249	        md_path = library_root / md_rel
   250	        if not md_path.exists():
   251	            continue
   252	        fm = _read_frontmatter(md_path)
   253	        if fm is None:
   254	            continue
   255	        # Reconcile frontmatter-authoritative fields
   256	        # review_status and review_flags are frontmatter-authoritative so
   257	        # manual edits (e.g. setting review_status: reviewed) are respected.
   258	        #
   259	        # NOT reconciled (converter-authoritative, recomputed on each conversion):
   260	        # - extraction_confidence: derived from evidence by assess_review_state()
   261	        # - grounding_summary: derived from evidence by _compute_grounding_summary()
   262	        # These are intentionally excluded; the registry retains the last-computed
   263	        # values and frontmatter is the source of truth only after re-conversion.
   264	        authoritative = [
   265	            "title", "type", "subtype", "client", "engagement", "authority",
   266	            "curation_level", "review_status", "review_flags",
   267	        ]
   268	        for field_name in authoritative:
   269	            if field_name in fm:
   270	                fm_val = fm[field_name]
   271	                reg_val = entry_data.get(field_name)
   272	                if fm_val != reg_val:
   273	                    entry_data[field_name] = fm_val
   274	                    changed += 1
   275	            else:
   276	                # Key absent from frontmatter = explicit deletion
   277	                if entry_data.get(field_name) is not None:
   278	                    entry_data[field_name] = None
   279	                    changed += 1
   280	        source_field = None
   281	        if "source_transcript" in fm:
   282	            source_field = "source_transcript"
   283	        elif "source" in fm:
   284	            source_field = "source"
   285	        if source_field is not None:
   286	            source_val = fm.get(source_field)
   287	            if source_val != entry_data.get("source_relative_path"):
   288	                entry_data["source_relative_path"] = source_val
   289	                changed += 1
   290	        # Also reconcile the ID if frontmatter has one
   291	        fm_id = fm.get("id")
   292	        if fm_id and fm_id != deck_id:
   293	            data["decks"][fm_id] = entry_data
   294	            entry_data["id"] = fm_id
   295	            del data["decks"][deck_id]
   296	            changed += 1
   297	    if changed:
   298	        logger.info("Registry reconciled: %d field(s) updated from frontmatter", changed)
   299	    return data

   401	def _atomic_write_json(path: Path, data: dict) -> None:
   402	    """Write JSON atomically with file locking.
   403	
   404	    Uses an advisory lock (``.lock`` file) to prevent concurrent writers
   405	    from racing, and cleans up the temp file on write failures
   406	    (disk-full, read-only filesystem, etc.).
   407	    """
   408	    path.parent.mkdir(parents=True, exist_ok=True)
   409	    lock_path = path.with_suffix(".lock")
   410	    tmp_path = path.with_suffix(".tmp")
   411	
   412	    try:
   413	        lock_fd = open(lock_path, "w")
   414	        fcntl.flock(lock_fd, fcntl.LOCK_EX)
   415	        try:
   416	            tmp_path.write_text(json.dumps(data, indent=2))
   417	            tmp_path.rename(path)
   418	        except OSError as e:
   419	            # Clean up orphan temp file on disk-full / read-only
   420	            try:
   421	                tmp_path.unlink(missing_ok=True)
   422	            except OSError:
   423	                pass
   424	            raise OSError(
   425	                f"Failed to write registry {path}: {e}. "
   426	                f"Check disk space and permissions."
   427	            ) from e
   428	        finally:
   429	            fcntl.flock(lock_fd, fcntl.LOCK_UN)
   430	            lock_fd.close()
   431	    except OSError:
   432	        # If we can't even open the lock file, try without locking
   433	        # (better than crashing, and the atomic rename still helps)
   434	        if 'lock_fd' not in dir():
   435	            raise

   438	def _read_frontmatter(md_path: Path) -> Optional[dict]:
   439	    """Read YAML frontmatter from a markdown file.
   440	
   441	    Returns None if the file has no valid frontmatter.
   442	    """
   443	    try:
   444	        content = md_path.read_text()
   445	        lines = content.split("\n")
   446	        if not lines or lines[0].strip() != "---":
   447	            return None
   448	        end_idx = None
   449	        for i, line in enumerate(lines[1:], start=1):
   450	            if line.strip() == "---":
   451	                end_idx = i
   452	                break
   453	        if end_idx is None:
   454	            return None
   455	        yaml_block = "\n".join(lines[1:end_idx])
   456	        result = yaml_lib.safe_load(yaml_block)
   457	        if not isinstance(result, dict):
   458	            return None
   459	        return result
   460	    except (yaml_lib.YAMLError, OSError):
   461	        return None
