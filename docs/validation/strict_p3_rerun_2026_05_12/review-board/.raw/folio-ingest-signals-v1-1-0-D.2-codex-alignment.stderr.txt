   108	def compute_speaker_stats(
   109	    source_text: str | Iterable[Turn],
   110	    *,
   111	    speaker_aliases: dict[str, str] | None = None,
   112	) -> SpeakerStats | None:
   113	    """Compute deterministic speaker stats.
   114	
   115	    Returns ``None`` when fewer than two timestamped speaker turns are
   116	    available. A single speaker can still be valid if there are multiple turns;
   117	    completely unstructured notes simply skip analytics.
   118	    """
   119	
   120	    turns = list(parse_turns(source_text) if isinstance(source_text, str) else source_text)
   121	    if speaker_aliases:
   122	        turns = [
   123	            replace(
   124	                turn,
   125	                speaker=_canonical_speaker(turn.speaker, speaker_aliases),
   126	            )
   127	            for turn in turns
   128	        ]
   129	    turns = [turn for turn in turns if turn.text.strip()]
   130	    if not turns:
   131	        return None
   132	
   133	    timed_turns = _with_inferred_end_times(turns)
   134	    totals_by_speaker: dict[str, list[tuple[Turn, int, int]]] = {}
   135	    for turn, word_count, duration_seconds in timed_turns:
   136	        totals_by_speaker.setdefault(turn.speaker, []).append(
   137	            (turn, word_count, duration_seconds)
   138	        )
   139	
   140	    total_words = sum(word_count for _, word_count, _ in timed_turns)
   141	    if total_words <= 0:
   142	        return None
   143	    first_start = min(turn.start_seconds for turn, _, _ in timed_turns)
   144	    last_end = max(turn.end_seconds or turn.start_seconds for turn, _, _ in timed_turns)
   145	    total_duration_seconds = max(1, round(last_end - first_start))
   146	    total_talk_seconds = max(1, sum(duration for _, _, duration in timed_turns))
   147	
   148	    per_speaker: dict[str, TurnStats] = {}
   149	    for speaker, speaker_turns in totals_by_speaker.items():
   150	        speaker_word_counts = [word_count for _, word_count, _ in speaker_turns]
   151	        word_count = sum(speaker_word_counts)
   152	        talk_time_seconds = sum(duration for _, _, duration in speaker_turns)
   153	        longest_index, longest_turn_words = max(
   154	            enumerate(speaker_word_counts),
   155	            key=lambda item: item[1],
   156	        )

   162	            speaker=speaker,
   163	            word_count=word_count,
   164	            word_share=round(word_count / total_words, 4),
   165	            turn_count=turn_count,

   173	        )
